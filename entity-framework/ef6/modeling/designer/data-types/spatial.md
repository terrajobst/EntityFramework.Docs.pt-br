<span data-ttu-id="b6ca5-164">O código adiciona dois novos objetos University ao contexto.</span><span class="sxs-lookup"><span data-stu-id="b6ca5-164">The code adds two new University objects to the context.</span></span> <span data-ttu-id="b6ca5-165">As propriedades espaciais são inicializadas usando o método DbGeography.
``` csharp
using (var context = new UniversityModelContainer())
{
    context.Universities.Add(new University()
    {
        Name = "Graphic Design Institute",
        Location = DbGeography.FromText("POINT(-122.336106 47.605049)"),
    });

    context.Universities.Add(new University()
    {
        Name = "School of Fine Art",
        Location = DbGeography.FromText("POINT(-122.335197 47.646711)"),
    });

    context.SaveChanges();

    var myLocation = DbGeography.FromText("POINT(-122.296623 47.640405)");

    var university = (from u in context.Universities
                                orderby u.Location.Distance(myLocation)
                                select u).FirstOrDefault();

    Console.WriteLine(
        "The closest University to you is: {0}.",
        university.Name);
}
```
